### 1.1 AI Assistant API Key Management

These scripts manage the pool of Gemini API keys used by the AI Assistant.

```sql
-- Create a new table to store and manage multiple Gemini API keys
-- This is the updated schema with new columns for default key and usage tracking.
CREATE TABLE public.ai_api_keys (
  id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
  api_key TEXT NOT NULL,
  status TEXT NOT NULL DEFAULT 'active' CHECK (status IN ('active', 'exhausted')),
  last_used_at TIMESTAMPTZ,
  failure_count INT NOT NULL DEFAULT 0,
  created_at TIMESTAMPTZ DEFAULT NOW() NOT NULL,
  is_default BOOLEAN NOT NULL DEFAULT false, -- NEW: To mark a key as the default
  call_count INT NOT NULL DEFAULT 0 -- NEW: To track how many times a key has been used
);

-- Add a unique index to ensure only one key can be the default.
CREATE UNIQUE INDEX one_default_key_idx ON public.ai_api_keys (is_default) WHERE is_default = true;


-- Enable Row Level Security for the new table
ALTER TABLE public.ai_api_keys ENABLE ROW LEVEL SECURITY;

-- Since this is an admin panel, we'll allow full access for authenticated users.
-- In a production multi-tenant app, you'd restrict this to specific admin roles.
CREATE POLICY "Allow full access for authenticated users"
ON public.ai_api_keys
FOR ALL
TO authenticated
USING (true)
WITH CHECK (true);

-- Create an index for faster lookups on active keys
CREATE INDEX idx_active_api_keys ON public.ai_api_keys(status, last_used_at);

-- UPDATED RPC function to get the next available API key.
-- It now prioritizes a default key if one is set and active.
-- Otherwise, it falls back to the smart rotation (least recently used) logic.
-- It also increments the call_count for the selected key.
CREATE OR REPLACE FUNCTION get_next_api_key()
RETURNS TABLE(key_id BIGINT, key_value TEXT) AS $$
DECLARE
  default_key_record RECORD;
  rotated_key_record RECORD;
BEGIN
  -- 1. Check if a default key is configured
  SELECT id, api_key, status INTO default_key_record
  FROM public.ai_api_keys
  WHERE is_default = true
  LIMIT 1;

  IF FOUND THEN
    -- 2. A default key exists. We MUST use it or fail.
    IF default_key_record.status = 'active' THEN
      -- The default key is active, so we use it.
      UPDATE public.ai_api_keys
      SET 
        last_used_at = NOW(),
        call_count = call_count + 1
      WHERE id = default_key_record.id;

      RETURN QUERY SELECT default_key_record.id, default_key_record.api_key;
    ELSE
      -- The default key is set but not active (exhausted). Return nothing to signal failure.
      RETURN;
    END IF;
  ELSE
    -- 3. No default key is set. Fall back to smart rotation.
    SELECT id, api_key INTO rotated_key_record
    FROM public.ai_api_keys
    WHERE status = 'active'
    ORDER BY last_used_at ASC NULLS FIRST
    LIMIT 1
    FOR UPDATE SKIP LOCKED;

    IF FOUND THEN
      -- A rotatable key was found. Use it.
      UPDATE public.ai_api_keys
      SET 
        last_used_at = NOW(),
        call_count = call_count + 1
      WHERE id = rotated_key_record.id;
      
      RETURN QUERY SELECT rotated_key_record.id, rotated_key_record.api_key;
    END IF;
  END IF;
END;
$$ LANGUAGE plpgsql;


-- Create an RPC function to mark a specific API key as exhausted
CREATE OR REPLACE FUNCTION mark_api_key_exhausted(p_key_id BIGINT)
RETURNS VOID AS $$
BEGIN
  UPDATE public.ai_api_keys
  SET 
    status = 'exhausted',
    failure_count = failure_count + 1
  WHERE id = p_key_id;
END;
$$ LANGUAGE plpgsql;

-- NEW RPC function to set a key as the default
CREATE OR REPLACE FUNCTION set_default_api_key(p_key_id BIGINT)
RETURNS VOID AS $$
BEGIN
  -- First, remove default status from all other keys
  UPDATE public.ai_api_keys
  SET is_default = false
  WHERE id <> p_key_id;

  -- Then, set the specified key as the default
  UPDATE public.ai_api_keys
  SET is_default = true
  WHERE id = p_key_id;
END;
$$ LANGUAGE plpgsql;

-- NEW RPC function to restore smart rotation by removing any default key
CREATE OR REPLACE FUNCTION restore_default_api_keys()
RETURNS VOID AS $$
BEGIN
  UPDATE public.ai_api_keys
  SET is_default = false
  WHERE is_default = true;
END;
$$ LANGUAGE plpgsql;

-- NEW RPC function to reset a single key's status to 'active'
CREATE OR REPLACE FUNCTION reset_api_key_status(p_key_id BIGINT)
RETURNS VOID AS $$
BEGIN
  UPDATE public.ai_api_keys
  SET
    status = 'active',
    failure_count = 0
  WHERE id = p_key_id;
END;
$$ LANGUAGE plpgsql;

```

### 1.2 AI Chat History

This section defines the table and functions for storing and retrieving conversations with the AI Assistant.

```sql
-- Create the table to store AI chat history
CREATE TABLE public.ai_chat_history (
  id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
  created_at TIMESTAMPTZ DEFAULT NOW() NOT NULL,
  session_id UUID NOT NULL,
  role TEXT NOT NULL, -- 'user' or 'model'
  content JSONB NOT NULL -- The message content in JSON format
);

-- Enable Row Level Security
ALTER TABLE public.ai_chat_history ENABLE ROW LEVEL SECURITY;

-- Create policies to allow users to access their own data
-- For this admin panel, we'll allow all authenticated users to do everything.
CREATE POLICY "Allow all access for authenticated users"
ON public.ai_chat_history
FOR ALL
TO authenticated
USING (TRUE)
WITH CHECK (TRUE);

-- Add an index for faster lookups by session_id
CREATE INDEX idx_ai_chat_history_session_id ON public.ai_chat_history(session_id);

-- Create RPC function to get chat sessions with their title and last message time
CREATE OR REPLACE FUNCTION get_chat_sessions()
RETURNS TABLE(session_id UUID, title TEXT, last_message_at TIMESTAMPTZ) AS $$
BEGIN
  RETURN QUERY
  WITH session_first_message AS (
    SELECT
      s.session_id,
      (s.content -> 'parts' -> 0 ->> 'text') AS title,
      ROW_NUMBER() OVER(PARTITION BY s.session_id ORDER BY s.created_at ASC) as rn
    FROM public.ai_chat_history s
    WHERE s.role = 'user'
  ),
  session_last_message AS (
    SELECT
      s.session_id,
      MAX(s.created_at) as last_message_at
    FROM public.ai_chat_history s
    GROUP BY s.session_id
  )
  SELECT
    slm.session_id,
    COLEASCE(sfm.title, 'New Chat') as title,
    slm.last_message_at
  FROM session_last_message slm
  LEFT JOIN session_first_message sfm ON slm.session_id = sfm.session_id AND sfm.rn = 1
  ORDER BY slm.last_message_at DESC;
END;
$$ LANGUAGE plpgsql;
```

### 1.4 Database & Function Analytics (NEW)

These functions allow the dashboard to show table activity (Inserts/Updates) and Edge Function health (Success/Failure).

```sql
-- 1. TABLE ANALYTICS
-- Create RPC function to get database usage stats from pg_stat_user_tables
-- Run this on BOTH Main App and Agent Database to see stats for both.
CREATE OR REPLACE FUNCTION get_database_analytics()
RETURNS TABLE(
  table_name text,
  live_rows bigint,
  total_inserts bigint,
  total_updates bigint,
  total_deletes bigint,
  last_used timestamptz
) AS $$
DECLARE
  r RECORD;
  max_time timestamptz;
  col_name text;
BEGIN
  FOR r IN 
    SELECT relname::text as tname, n_live_tup, n_tup_ins, n_tup_upd, n_tup_del
    FROM pg_stat_user_tables
    WHERE schemaname = 'public'
    ORDER BY n_tup_ins DESC
  LOOP
    table_name := r.tname;
    live_rows := r.n_live_tup;
    total_inserts := r.n_tup_ins;
    total_updates := r.n_tup_upd;
    total_deletes := r.n_tup_del;
    last_used := NULL;
    
    -- Try to find a timestamp column
    SELECT column_name INTO col_name
    FROM information_schema.columns
    WHERE table_schema = 'public' AND information_schema.columns.table_name = r.tname
    AND column_name IN ('updated_at', 'created_at', 'timestamp', 'last_sign_in_at', 'created')
    ORDER BY 
      CASE column_name 
        WHEN 'updated_at' THEN 1 
        WHEN 'last_sign_in_at' THEN 2
        WHEN 'created_at' THEN 3 
        WHEN 'created' THEN 4
        WHEN 'timestamp' THEN 5 
        ELSE 6 
      END
    LIMIT 1;

    IF col_name IS NOT NULL THEN
      BEGIN
        EXECUTE format('SELECT max(%I) FROM public.%I', col_name, r.tname) INTO max_time;
        last_used := max_time;
      EXCEPTION WHEN OTHERS THEN
        last_used := NULL;
      END;
    END IF;

    RETURN NEXT;
  END LOOP;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

GRANT EXECUTE ON FUNCTION get_database_analytics() TO authenticated;
GRANT EXECUTE ON FUNCTION get_database_analytics() TO service_role;


-- 2. EDGE FUNCTION ANALYTICS
-- This RPC aggregates data from the log tables to show function health.

-- >>> RUN THIS IN MAIN APP DATABASE <<<
CREATE OR REPLACE FUNCTION get_function_stats()
RETURNS TABLE(
  function_name text,
  total_calls bigint,
  success_count bigint,
  error_count bigint,
  last_run timestamptz
) AS $$
BEGIN
  RETURN QUERY
  SELECT
    'News Updater'::text as function_name,
    COUNT(*) as total_calls,
    COUNT(*) FILTER (WHERE status = 'SUCCESS') as success_count,
    COUNT(*) FILTER (WHERE status != 'SUCCESS') as error_count,
    MAX(created_at) as last_run
  FROM update_news_logs;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

GRANT EXECUTE ON FUNCTION get_function_stats() TO authenticated;
GRANT EXECUTE ON FUNCTION get_function_stats() TO service_role;
```

### 2.1 Data & Sequence Reset Functions

These functions are specific to the `groq_agent_logs` table in the Agent database.

```sql
-- =================================================================
--  FUNCTION 1: Reset ID Sequence for groq_agent_logs
--  Description: Resets the auto-incrementing 'id' for the table.
--               The next log will have id = 1.
-- =================================================================
CREATE OR REPLACE FUNCTION reset_groq_agent_logs_id_sequence()
RETURNS void AS $$
BEGIN
  -- Sets the next value for the 'id' column's sequence to 1
  ALTER TABLE public.groq_agent_logs ALTER COLUMN id RESTART WITH 1;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

-- Permissions: Only allow service_role (backend) to execute
REVOKE EXECUTE ON FUNCTION reset_groq_agent_logs_id_sequence() FROM PUBLIC;
GRANT EXECUTE ON FUNCTION reset_groq_agent_logs_id_sequence() TO service_role;


-- =================================================================
--  FUNCTION 2: Truncate (Delete all data from) groq_agent_logs
--  Description: Permanently deletes all rows from the table.
--               This is faster than DELETE and also resets the ID sequence.
-- =================================================================
CREATE OR REPLACE FUNCTION truncate_table_groq_agent_logs()
RETURNS void AS $$
BEGIN
  -- Wipes all data from the table and resets the ID sequence
  TRUNCATE TABLE public.groq_agent_logs RESTART IDENTITY CASCADE;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

-- Permissions: Only allow service_role (backend) to execute
REVOKE EXECUTE ON FUNCTION truncate_table_groq_agent_logs() FROM PUBLIC;
GRANT EXECUTE ON FUNCTION truncate_table_groq_agent_logs() TO service_role;
```
